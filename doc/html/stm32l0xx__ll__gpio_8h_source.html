669__STATIC_INLINE void LL_GPIO_LockPin(GPIO_TypeDef *GPIOx, uint32_t PinMask)
670{
671 __IO uint32_t temp;
672 WRITE_REG(GPIOx->LCKR, GPIO_LCKR_LCKK | PinMask);
673 WRITE_REG(GPIOx->LCKR, PinMask);
674 WRITE_REG(GPIOx->LCKR, GPIO_LCKR_LCKK | PinMask);
675 /* Read LCKK register. This read is mandatory to complete key lock sequence */
676 temp = READ_REG(GPIOx->LCKR);
677 (void) temp;
678}
