220__STATIC_INLINE void LL_InitTick(uint32_t HCLKFrequency, uint32_t Ticks)
221{
222 /* Configure the SysTick to have interrupt in 1ms time base */
223 SysTick->LOAD = (uint32_t)((HCLKFrequency / Ticks) - 1UL); /* set reload register */
224 SysTick->VAL = 0UL; /* Load the SysTick Counter Value */
225 SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk |
226 SysTick_CTRL_ENABLE_Msk; /* Enable the Systick Timer */
227}
