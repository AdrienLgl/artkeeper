1612 uint32_t brrtemp;

1615 {
1616 usartdiv = (uint16_t)(__LL_USART_DIV_SAMPLING8(PeriphClk, BaudRate));
1617 brrtemp = usartdiv & 0xFFF0U;
1618 brrtemp |= (uint16_t)((usartdiv & (uint16_t)0x000FU) >> 1U);
1619 USARTx->BRR = brrtemp;
1620 }
1621 else
1622 {
1623 USARTx->BRR = (uint16_t)(__LL_USART_DIV_SAMPLING16(PeriphClk, BaudRate));

1640__STATIC_INLINE uint32_t LL_USART_GetBaudRate(USART_TypeDef *USARTx, uint32_t PeriphClk, uint32_t OverSampling)
1641{
1642 uint32_t usartdiv;
1643 uint32_t brrresult = 0x0U;
1644
1645 usartdiv = USARTx->BRR;
1646
1647 if (usartdiv == 0U)
1648 {
1649 /* Do not perform a division by 0 */
1650 }
1651 else if (OverSampling == LL_USART_OVERSAMPLING_8)
1652 {
1653 usartdiv = (uint16_t)((usartdiv & 0xFFF0U) | ((usartdiv & 0x0007U) << 1U)) ;
1654 if (usartdiv != 0U)
1655 {
1656 brrresult = (PeriphClk * 2U) / usartdiv;
1657 }
1658 }
1659 else
1660 {
1661 if ((usartdiv & 0xFFFFU) != 0U)
1662 {
1663 brrresult = PeriphClk / usartdiv;
1664 }
1665 }
